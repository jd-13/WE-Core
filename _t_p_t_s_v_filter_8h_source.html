114 template <typename T>
115 void TPTSVFilter<T>::processBlock(T* inSamples, size_t numSamples) {
116
117 if (_mode != Parameters::FILTER_MODE.BYPASS) {
118
119 for (size_t idx {0}; idx < numSamples; idx++) {
120 const T sample {inSamples[idx]};
121
122 const T yH {static_cast<T>(_h * (sample - (1.0f / _q + _g) * _s1 - _s2))};
123
124 const T yB {static_cast<T>(_g * yH + _s1)};
125 _s1 = _g * yH + yB;
126
127 const T yL {static_cast<T>(_g * yB + _s2)};
128 _s2 = _g * yB + yL;
129
130 switch (_mode) {
131 case Parameters::ModeParameter::PEAK:
132 inSamples[idx] = yB * static_cast<T>(_gain);
133 break;
134
135 case Parameters::ModeParameter::HIGHPASS:
136 inSamples[idx] = yH * static_cast<T>(_gain);
137 break;
138
139 default:
140 inSamples[idx] = yL * static_cast<T>(_gain);
141 break;
142 }
143 }
144 }
145 }

165 template <typename T>
166 void TPTSVFilter<T>::_calculateCoefficients() {
167 _g = static_cast<T>(std::tan(CoreMath::DOUBLE_PI * _cutoffHz / _sampleRate));
168 _h = static_cast<T>(1.0 / (1 + _g / _q + _g * _g));
