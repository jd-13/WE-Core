200  template <typename SampleType>
201  void MONSTRBand<SampleType>::setLowCutoff(double val) {
202  _lowCutoffHz = Parameters::CROSSOVER_FREQUENCY.BoundsCheck(val);
203  _lowCut1.setup(FILTER_ORDER, _sampleRate, _lowCutoffHz);
204  _lowCut2.setup(FILTER_ORDER, _sampleRate, _lowCutoffHz);
205 
206  // Move the high cutoff up if necessary as they shouldn't swap places
207  if (_lowCutoffHz > _highCutoffHz) {

239  template <typename SampleType>
240  void MONSTRBand<SampleType>::process2in2out(SampleType* leftSample,
241  SampleType* rightSample,
242  size_t numSamples) {
243 
244  if (_isMuted) {
245  // Muted - set the output to 0 for this band
246  for (size_t sampleIndex {0}; sampleIndex < numSamples; sampleIndex++) {
247  leftSample[sampleIndex] = 0;
248  rightSample[sampleIndex] = 0;
249  }
250  } else {
251  // Apply the filtering before processing
252  _filterSamples(leftSample, rightSample, static_cast<int>(numSamples));
253 
254  if (_isActive) {
255  if (_processor != nullptr) {
256  _processor->process2in2out(leftSample, rightSample, numSamples);
257  }
258  }
259  }
260  }

270  template <typename SampleType>
271  void MONSTRBand<SampleType>::_filterSamples(SampleType* inLeftSamples, SampleType* inRightSamples, int numSamples) {
272  SampleType* channelsArray[2];
273 
274  channelsArray[0] = inLeftSamples;
275  channelsArray[1] = inRightSamples;
276 
277  if (_bandType == BandType::LOWER) {
278  _highCut1.process(numSamples, channelsArray);
279  _highCut2.process(numSamples, channelsArray);
280  } else if (_bandType == BandType::UPPER) {
281  _lowCut1.process(numSamples, channelsArray);
282  _lowCut2.process(numSamples, channelsArray);
283  } else {
284  _lowCut1.process(numSamples, channelsArray);
285  _lowCut2.process(numSamples, channelsArray);
286  _highCut1.process(numSamples, channelsArray);
287  _highCut2.process(numSamples, channelsArray);
288  }
289  }
