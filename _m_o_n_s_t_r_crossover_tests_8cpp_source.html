334
335 WECore::MONSTR::MONSTRCrossover<double> mCrossover;
336 mCrossover.addBand();
337 mCrossover.setIsActive(3, false);
338 mCrossover.setIsMuted(3, true);
339 mCrossover.setCrossoverFrequency(2, 15000);
340
341 auto processorStrong = std::make_shared<WECore::StereoWidth::StereoWidthProcessor<double>>();
342 mCrossover.setEffectsProcessor(3, processorStrong);
343
344 // Take a weak pointer to the effects processor and reset the original shared (strong)
345 // pointer - this will be expired if the shared pointer held by the band is correctly reset
346 std::weak_ptr<WECore::StereoWidth::StereoWidthProcessor<double>> processorWeak = processorStrong;
347 processorStrong.reset();
348
349 WHEN("The 4th band is removed and added again") {
350 mCrossover.removeBand();
351 mCrossover.addBand();
352
353 THEN("The band has the default parameters") {

355 CHECK(mCrossover.getIsMuted(3) == false);
