25SCENARIO("RichterLFOPair: Parameters can be set and retrieved correctly") {
26 GIVEN("A new RichterLFOPair object") {
27 WECore::Richter::RichterLFOPair mLFOPair;

32 CHECK(mLFOPair.LFO.getPhaseSyncSwitch() == true);
33 CHECK(mLFOPair.LFO.getTempoSyncSwitch() == false);
34 CHECK(mLFOPair.LFO.getInvertSwitch() == false);
35 CHECK(mLFOPair.LFO.getWave() == 1);
36 CHECK(mLFOPair.LFO.getDepth() == Approx(0.5));
37 CHECK(mLFOPair.LFO.getDepthMod() == Approx(0.0));
38 CHECK(mLFOPair.LFO.getFreq() == Approx(2.0));
39 CHECK(mLFOPair.LFO.getFreqMod() == Approx(0.0));
40 CHECK(mLFOPair.LFO.getManualPhase() == Approx(0.0));
41 CHECK(mLFOPair.LFO.getTempoNumer() == Approx(1.0));
42 CHECK(mLFOPair.LFO.getTempoDenom() == Approx(1.0));
43
44 CHECK(mLFOPair.MOD->getBypassSwitch() == false);
45 CHECK(mLFOPair.MOD->getPhaseSyncSwitch() == true);
46 CHECK(mLFOPair.MOD->getTempoSyncSwitch() == false);
47 CHECK(mLFOPair.MOD->getInvertSwitch() == false);
48 CHECK(mLFOPair.MOD->getWave() == 1);
49 CHECK(mLFOPair.MOD->getDepth() == Approx(0.5));
50 CHECK(mLFOPair.MOD->getFreq() == Approx(2.0));
51 CHECK(mLFOPair.MOD->getManualPhase() == Approx(0.0));
