28SCENARIO("MONSTRBand: Parameters can be set and retrieved correctly") {
29 GIVEN("A new MONSTRBand object") {
30 WECore::MONSTR::MONSTRBand<double> mBand(WECore::MONSTR::BandType::MIDDLE);
31
32 WHEN("Nothing is changed") {
33 THEN("Parameters have their default values") {
34 CHECK(mBand.getIsActive() == true);
35 CHECK(mBand.getIsMuted() == false);
36 CHECK(mBand.getLowCutoff() == Approx(100.0));
37 CHECK(mBand.getHighCutoff() == Approx(5000.0));
38 }
39 }
40
41 WHEN("All parameters are changed to unique values") {
42
43 mBand.setIsActive(false);
44 mBand.setIsMuted(false);
45 mBand.setLowCutoff(201);
46 mBand.setHighCutoff(6001);
47
48 THEN("They all get their correct unique values") {
49 CHECK(mBand.getIsActive() == false);
50 CHECK(mBand.getIsMuted() == false);
51 CHECK(mBand.getLowCutoff() == Approx(201.0));
52 CHECK(mBand.getHighCutoff() == Approx(6001.0));
53 }
54 }
55 }

58SCENARIO("MONSTRBand: Parameters enforce their bounds correctly") {
59 GIVEN("A new MONSTRBand object") {
60 WECore::MONSTR::MONSTRBand<double> mBand(WECore::MONSTR::BandType::MIDDLE);
61
62 WHEN("All parameter values are too low") {
63 mBand.setLowCutoff(39);
64 mBand.setHighCutoff(39);
65
66 THEN("Parameters enforce their lower bounds") {
67 CHECK(mBand.getLowCutoff() == Approx(40.0));
68 CHECK(mBand.getHighCutoff() == Approx(40.0));
69 }
70 }
71
72 WHEN("All parameter values are too high") {
73 mBand.setLowCutoff(20000);
74 mBand.setHighCutoff(20000);
75
76 THEN("Parameters enforce their upper bounds") {
77 CHECK(mBand.getLowCutoff() == Approx(19500.0));
78 CHECK(mBand.getHighCutoff() == Approx(19500.0));
79 }
80 }
81
82 WHEN("The high cutoff is set below the lower one") {
83 mBand.setLowCutoff(4000);
84 mBand.setHighCutoff(2000);
85
86 THEN("Both frequencies move to the lower value") {
87 CHECK(mBand.getLowCutoff() == Approx(2000.0));
88 CHECK(mBand.getHighCutoff() == Approx(2000.0));
89 }
90 }
91
92 WHEN("The lower cutoff is set above the higher one") {
93 mBand.setHighCutoff(2000);
94 mBand.setLowCutoff(4000);
95
96 THEN("Both frequencies move to the higher value") {
97 CHECK(mBand.getLowCutoff() == Approx(4000.0));
98 CHECK(mBand.getHighCutoff() == Approx(4000.0));
99 }
100 }
101 }

105SCENARIO("MONSTRBand: Muted") {
106 GIVEN("A MONSTRBand and a buffer of sine samples") {
107 std::vector<double> leftBuffer(1024);
108 std::vector<double> rightBuffer(1024);
109
110 WECore::MONSTR::MONSTRBand<double> mBand(WECore::MONSTR::BandType::MIDDLE);
111
112 WHEN("The band is muted") {
113
114 mBand.setIsMuted(true);
115
116 // Fill the buffer
117 WECore::TestUtils::generateSine(leftBuffer, 44100, 1000);
118 std::copy(leftBuffer.begin(), leftBuffer.end() , rightBuffer.begin());
119
120 // Do processing
121 mBand.process2in2out(&leftBuffer[0], &rightBuffer[0], leftBuffer.size());
122
123 THEN("The output is silence") {
124 for (size_t index {0}; index < leftBuffer.size(); index++) {
125 CHECK(leftBuffer[index] == Approx(0.0));
126 CHECK(rightBuffer[index] == Approx(0.0));
127 }
128 }
129 }
