278 template <typename SampleType>
279 void MONSTRCrossover<SampleType>::setCrossoverFrequency(size_t index, double val) {
280
281 if (index < _bands.size() - 1) {
282
283 // Set the crossover frequency
284 val = Parameters::CROSSOVER_FREQUENCY.BoundsCheck(val);
285 _bands[index].band.setHighCutoff(val);
286 _bands[index + 1].band.setLowCutoff(val);
287
288 // Make sure the crossover frequencies are still in the correct order
289 for (size_t otherCrossoverIndex {0}; otherCrossoverIndex < _bands.size() - 1; otherCrossoverIndex++) {
290
291 const double otherCrossoverFrequency {getCrossoverFrequency(otherCrossoverIndex)};
292
293 const bool needsCrossoverUpdate {
294 // We've moved the crossover frequency of index below another one that should be
295 // below it - move the other one to the new value
296 (val < otherCrossoverFrequency && otherCrossoverIndex < index) ||
297
298 // We've moved the crossover frequency of index above another one that should be
299 // above it - move the other one to the new value
300 (otherCrossoverFrequency > val && index > otherCrossoverIndex)
301 };
302
303 if (needsCrossoverUpdate) {
304 // We've moved the crossover frequency of index below another one that should be
305 // below it - move the other one to the new value
306 setCrossoverFrequency(otherCrossoverIndex, val);
307
308 }
309 }
310 }

388 template <typename SampleType>
389 void MONSTRCrossover<SampleType>::addBand() {
390
391 if (static_cast<int>(_numBands) < Parameters::NUM_BANDS.maxValue) {
392
393 // Convert the current highest band to a middle band
394 _bands[_numBands - 1].band.setBandType(BandType::MIDDLE);
395
396 // Set the next band up to be an upper band
397 _bands[_numBands].band.setBandType(BandType::UPPER);
398
399 const double oldHighestCrossover {getCrossoverFrequency(_numBands - 2)};
400
401 if (oldHighestCrossover < Parameters::CROSSOVER_FREQUENCY.maxValue) {
402 // The old highest crossover frequency is below the maximum, insert the new one halfway
403 // between it and the maximum
404 const double topBandWidth {Parameters::CROSSOVER_FREQUENCY.maxValue - oldHighestCrossover};
405 setCrossoverFrequency(_numBands - 1, oldHighestCrossover + (topBandWidth / 2));
406 } else {
407 // The old highest crossover is at the maximum, move it halfway down to the one below
408 // and place the new one at the maximum
409 const double secondHighestCrossover {getCrossoverFrequency(_numBands - 3)};
410 const double bandWidth {Parameters::CROSSOVER_FREQUENCY.maxValue - secondHighestCrossover};
411
412 setCrossoverFrequency(_numBands - 2, secondHighestCrossover + (bandWidth / 2));
413 setCrossoverFrequency(_numBands - 1, Parameters::CROSSOVER_FREQUENCY.maxValue);
414 }
415
416 _numBands++;
417 }
418
419 reset();
420 }

422 template <typename SampleType>
423 void MONSTRCrossover<SampleType>::removeBand() {
424
425 if (static_cast<int>(_numBands) > Parameters::NUM_BANDS.minValue) {
426 // Make sure the band's parameters are reset to their default values
427 setIsActive(_numBands - 1, true);
428 setIsMuted(_numBands - 1, false);
429 setEffectsProcessor(_numBands - 1, nullptr);
430
431 // Decrement the counter and make the new highest band the upper band
432 _numBands--;
433 _bands[_numBands - 1].band.setBandType(BandType::UPPER);
434 }
435
436 reset();
437 }

439 template <typename SampleType>
440 void MONSTRCrossover<SampleType>::Process2in2out(SampleType* leftSample,
441 SampleType* rightSample,
442 size_t numSamples) {
443
444 // If the buffer we've been passed is bigger than our static internal buffer, then we need
445 // to break it into chunks
446 const size_t numBuffersRequired {static_cast<size_t>(
447 std::ceil(static_cast<double>(numSamples) / INTERNAL_BUFFER_SIZE)
448 )};
449
450 for (size_t bufferNumber {0}; bufferNumber < numBuffersRequired; bufferNumber++) {
451
452 // Calculate how many samples need to be processed in this chunk
453 const size_t numSamplesRemaining {numSamples - (bufferNumber * INTERNAL_BUFFER_SIZE)};
454 const size_t numSamplesToCopy {
455 std::min(numSamplesRemaining, static_cast<size_t>(INTERNAL_BUFFER_SIZE))
456 };
457
458 SampleType* const leftBufferInputStart {&leftSample[bufferNumber * INTERNAL_BUFFER_SIZE]};
459 SampleType* const rightBufferInputStart {&rightSample[bufferNumber * INTERNAL_BUFFER_SIZE]};
460
461 // Populate the band specific buffers, and do the processing
462 for (size_t bandIndex {0}; bandIndex < _numBands; bandIndex++) {
463
464 // Only do processing no bands are soloed or if this band is soloed
465 if (_numBandsSoloed == 0 || _bands[bandIndex].isSoloed) {
466 std::copy(leftBufferInputStart,
467 &leftBufferInputStart[numSamplesToCopy],
468 _bands[bandIndex].leftBuffer);
469
470 std::copy(rightBufferInputStart,
471 &rightBufferInputStart[numSamplesToCopy],
472 _bands[bandIndex].rightBuffer);
473
474 _bands[bandIndex].band.process2in2out(_bands[bandIndex].leftBuffer,
475 _bands[bandIndex].rightBuffer,
476 numSamplesToCopy);
477 }
478 }
479
480 // Combine the output from each band, and write to output
481 for (size_t sampleIndex {0}; sampleIndex < numSamplesToCopy; sampleIndex++) {
482
483 // Reset the sample to zero first
484 leftBufferInputStart[sampleIndex] = 0;
485 rightBufferInputStart[sampleIndex] = 0;
486
487 for (size_t bandIndex {0}; bandIndex < _numBands; bandIndex++) {
488
489 // Only do processing no bands are soloed or if this band is soloed
490 if (_numBandsSoloed == 0 || _bands[bandIndex].isSoloed) {
491 leftBufferInputStart[sampleIndex] += _bands[bandIndex].leftBuffer[sampleIndex];
492 rightBufferInputStart[sampleIndex] += _bands[bandIndex].rightBuffer[sampleIndex];
493 }
494 }
495 }
496 }
497 }
