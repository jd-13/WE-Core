228 void RichterLFO::_calcPhaseOffset(double timeInSeconds) {
229
230 // The phase offset applied by the playhead's start position needs to be calculated only
231 // when playback initially starts, and not for any subsequent buffers until playback stops
232 // and starts again
233 static int seekIndexOffset {0};
234 if (_needsSeekOffsetCalc) {
235 const double waveLength {1 / _calcFreq(0)};
236 const double waveTimePosition {std::fmod(timeInSeconds, waveLength)};
237
238 seekIndexOffset = static_cast<int>((waveTimePosition / waveLength) * Wavetables::SIZE);
239 _needsSeekOffsetCalc = false;
240 }
241
242 if (_phaseSyncSwitch) {
243 const int phaseIndexOffset {
244 static_cast<int>((_manualPhase / Parameters::PHASE.maxValue) * Wavetables::SIZE)
245 };
246 _indexOffset = seekIndexOffset + phaseIndexOffset;
247 } else {
248 _indexOffset = 0;
249 }
250 }

252 double RichterLFO::_calcFreq(double modAmount) {
253 // Calculate the frequency based on whether tempo sync is active
254
255 double freq {0};
256
257 if (_tempoSyncSwitch) {
258 freq = (_bpm / 60) * (_tempoDenom / _tempoNumer);
259 } else {
260 freq = _rawFreq + (_freqMod * (Parameters::FREQ.maxValue / 2) * modAmount);
261 }
262
263 return Parameters::FREQ.BoundsCheck(freq);
264 }
265
266 double RichterLFO::_calcLFOValue(double freq) {
267 const double samplesPerTremoloCycle {_sampleRate / freq};
268 const double scale {Wavetables::SIZE / samplesPerTremoloCycle};
269
270 // Calculate the current position within the wave table
271 _wavetablePosition = std::fmod(_wavetablePosition + scale, Wavetables::SIZE);
272
273 const int index {static_cast<int>(_wavetablePosition)};
274
275 return _waveArrayPointer[(index + _indexOffset) % Wavetables::SIZE];
276 }

278 double RichterLFO::_getNextOutputImpl(double inSample) {
279 // Get the mod amount to use, divide by 2 to reduce range to -0.5:0.5
280 // This is the only function that should advance the modulation state by calling getNextOutput()
281 const double modAmount {_modulationSource != nullptr ? _modulationSource->getNextOutput(inSample) / 2 : 0};
282
283 const double lfoValue {_calcLFOValue(_calcFreq(modAmount))};
284
285 // Calculate the depth value after modulation
286 const double depth {Parameters::DEPTH.BoundsCheck(
287 _rawDepth + (_depthMod * Parameters::DEPTH.maxValue * modAmount)
288 )};
289
290 if (_bypassSwitch) {
291 // Produce a value in the range -1:1, invert if needed
292 return (lfoValue * depth) * (_invertSwitch ? -1 : 1);
293 } else {
294 return 0;
295 }
296 }
