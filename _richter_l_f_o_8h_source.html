194 double _calcModValue(const std::vector<ModulationSourceWrapper<double>>& sources) const {
195 double retVal {0};
196
197 for (const ModulationSourceWrapper<double>& source : sources) {
198 retVal += source.source->getLastOutput() * source.amount;
199 }

349 void RichterLFO::_calcPhaseOffset(double timeInSeconds) {
350
351 // The phase offset applied by the playhead's start position needs to be calculated only
352 // when playback initially starts, and not for any subsequent buffers until playback stops
353 // and starts again
354 static int seekIndexOffset {0};
355 if (_needsSeekOffsetCalc) {
356 const double waveLength {1 / _calcFreq(0)};
357 const double waveTimePosition {std::fmod(timeInSeconds, waveLength)};
358
359 seekIndexOffset = static_cast<int>((waveTimePosition / waveLength) * Wavetables::SIZE);
360 _needsSeekOffsetCalc = false;
361 }
362
363 if (_phaseSyncSwitch) {
364 _indexOffset = seekIndexOffset;
365 } else {
366 _indexOffset = 0;
367 }
368 }

370 double RichterLFO::_calcFreq(double modAmount) {
371 // Calculate the frequency based on whether tempo sync is active
372 double freq {0};
373
374 if (_tempoSyncSwitch) {
375 freq = (_bpm / 60) * (_tempoDenom / _tempoNumer);
376 } else {
377 freq = _rawFreq + ((Parameters::FREQ.maxValue / 2) * modAmount);
378 }
379
380 freq = Parameters::FREQ.BoundsCheck(freq);
381 _modulatedFreqValue = freq;
382
383 return freq;
384 }

386 double RichterLFO::_calcLFOValue(double freq, double phaseModValue) {
387 const double samplesPerTremoloCycle {_sampleRate / freq};
388 const double scale {Wavetables::SIZE / samplesPerTremoloCycle};
389
390 // Calculate the current position within the wave table
391 _wavetablePosition = std::fmod(_wavetablePosition + scale, Wavetables::SIZE);
392
393 _modulatedPhaseValue = _manualPhase + (phaseModValue * Parameters::PHASE.maxValue);
394
395 const int phaseIndexOffset {
396 static_cast<int>((_modulatedPhaseValue / Parameters::PHASE.maxValue) * Wavetables::SIZE)
397 };
398
399 const int index {static_cast<int>(_wavetablePosition)};
400
401 const double wavetableValue {_waveArrayPointer[(index + _indexOffset + phaseIndexOffset) % Wavetables::SIZE]};
402
403 return _outputMode == Parameters::OUTPUTMODE.BIPOLAR ? wavetableValue : (wavetableValue + 1);
404 }

406 double RichterLFO::_getNextOutputImpl(double /*inSample*/) {
407 // Get the mod amount to use, divide by 2 to reduce range to -0.5:0.5
408 const double freqModValue {_calcModValue(_freqModulationSources) / 2};
409 const double depthModValue {_calcModValue(_depthModulationSources) / 2};
410 const double phaseModValue {_calcModValue(_phaseModulationSources) / 2};
411
412 const double lfoValue {_calcLFOValue(_calcFreq(freqModValue), phaseModValue)};
413
414 // Calculate the depth value after modulation
415 const double depth {Parameters::DEPTH.BoundsCheck(
416 _rawDepth + (Parameters::DEPTH.maxValue * depthModValue)
417 )};
418 _modulatedDepthValue = depth;
419
420 if (_bypassSwitch) {
421 // Produce a value in the range -1:1 (or 0:2), invert if needed
422 return (lfoValue * depth) * (_invertSwitch ? -1 : 1);
423 } else {
424 return 0;
425 }
426 }
