75 void TooltipLabelUpdater::start(juce::Label* targetLabel, juce::AudioProcessor::WrapperType pluginFormat, const juce::String& buildHash, bool isDemo, bool isPreRelease) {
76 _targetLabel = targetLabel;
77
78 _defaultString = JucePlugin_Name;
79 _defaultString += " ";
80 _defaultString += JucePlugin_VersionString;
81
82 // Format
83 _defaultString += " ";
84 _defaultString += juce::AudioProcessor::getWrapperTypeDescription(pluginFormat);
85
86 // OS
87 _defaultString += " ";
88#if _WIN32
89 _defaultString += "Win";
90#elif __APPLE__
91 _defaultString += "macOS";
92#elif __linux__
93 _defaultString += "Linux";
94#else
95 #error "Unknown OS"
96#endif
97
98 // Arch
99 _defaultString += " ";
100#if defined(__x86_64__) || defined(_M_AMD64)
101 _defaultString += "x86_64";
102#elif defined(__aarch64__) || defined(_M_ARM64)
103 _defaultString += "arm64";
104#else
105 #error "Unknown arch"
106#endif
107
108 // Build hash
109 _defaultString += " ";
110 _defaultString += buildHash;
111
112 // Demo
113 if (isDemo) {
114 _defaultString += " (DEMO)";
115 }
116
117 // Pre-release build
118 if (isPreRelease) {
119 _defaultString += " (PRE-RELEASE)";
120 }
121
122 _targetLabel->setText(_defaultString, juce::dontSendNotification);
123 }
124
125 void TooltipLabelUpdater::mouseEnter(const juce::MouseEvent& event) {
126 if (_targetLabel != nullptr) {
127 juce::TooltipClient* tooltipClient = dynamic_cast<juce::TooltipClient*>(event.eventComponent);
128
129 if (tooltipClient != nullptr) {
130 const juce::String displayString = tooltipClient->getTooltip().isEmpty() ? _defaultString : tooltipClient->getTooltip();
131 _targetLabel->setText(displayString, juce::dontSendNotification);
132 }
133 }
134 }
135
136 void TooltipLabelUpdater::mouseExit(const juce::MouseEvent& /*event*/) {
137 if (_targetLabel != nullptr) {
138 _targetLabel->setText(_defaultString, juce::dontSendNotification);
139 }
140 }
