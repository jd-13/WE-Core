129  double RichterLFOPair::_getNextOutputImpl(double /*inSample*/) {
130 
131  double retVal {1};
132 
133  // Always call getNextOutput regardless of bypassed state
134  const double tempGain {LFO.getNextOutput(0)};
135 
136  if (LFO.getBypassSwitch()) {
137  // The output of the LFO is a value in the range -1:1, we need to convert this into a
138  // gain in the range 0:1 and make sure the value is 1 when the depth is 0
139  retVal = (tempGain / 2) + (2 - LFO.getDepth()) / 2;
140  }
141 
142  return retVal;
143  }
