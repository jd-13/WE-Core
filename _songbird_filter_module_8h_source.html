385 template <typename T>
386 void SongbirdFilterModule<T>::Process1in1out(T* inSamples, size_t numSamples) {
387
388 for (size_t index {0}; index < numSamples; index++) {
389
390 // Figure out the modulation here. We have two ways to modulation between
391 // two formant filters.
392 // For MODMODE_BLEND we modulation the filter position to blend between the two filters.
393 // For MODMOD_FREQ we set the filter position to 0 so that we're only using filter 1,
394 // and then modulate the freqency of filter 1 between the two vowels
395 double blendFilterPosition {0};
396 const double modAmount {_modulationSource != nullptr ? _modulationSource->getNextOutput(inSamples[index]) : 0};
397 if (_modMode == Parameters::MODMODE_BLEND) {
398 blendFilterPosition = _filterPosition + modAmount;
399 } else {
400 blendFilterPosition = 0;
401 _setVowel1(_calcVowelForFreqMode(modAmount));
402 }
403
404 // Do the processing for each filter
405 const T originalInput {inSamples[index]};
406 const T filter1Out = _filter1Left.process(originalInput);
407 const T filter2Out = _filter2Left.process(originalInput);
408 const T filterAirOut = _filterAirLeft.process(originalInput);
409
410 // Write to output, applying filter position and mix level
411 inSamples[index] = (
412 originalInput * (1 - _mix)
413 + filter1Out * (1 - blendFilterPosition) * _mix
414 + filter2Out * blendFilterPosition * _mix
415 + filterAirOut * _airGain * _mix
416 )
417 * _outputGain;
418 }
419 }

421 template <typename T>
422 void SongbirdFilterModule<T>::Process2in2out(T* leftSamples,
423 T* rightSamples,
424 size_t numSamples) {
425
426 for (size_t index {0}; index < numSamples; index++) {
427
428 // Figure out the modulation here. We have two ways to modulation between
429 // two formant filters.
430 // For MODMODE_BLEND we modulation the filter position to blend between the two filters.
431 // For MODMOD_FREQ we set the filter position to 0 so that we're only using filter 1,
432 // and then modulate the freqency of filter 1 between the two vowels
433 double blendFilterPosition {0};
434 const double modAmount {
435 _modulationSource != nullptr ? _modulationSource->getNextOutput((leftSamples[index] + rightSamples[index]) / 2) : 0
436 };
437
438 if (_modMode == Parameters::MODMODE_BLEND) {
439 blendFilterPosition = _filterPosition + modAmount;
440 } else {
441 blendFilterPosition = 0;
442 _setVowel1(_calcVowelForFreqMode(modAmount));
443 }
444
445 // Do the processing for each filter
446 const T originalLeftIn {leftSamples[index]};
447 const T originalRightIn {rightSamples[index]};
448
449 const T filter1LeftOut = _filter1Left.process(originalLeftIn);
450 const T filter1RightOut = _filter1Right.process(originalRightIn);
451
452 const T filter2LeftOut = _filter2Left.process(originalLeftIn);
453 const T filter2RightOut = _filter2Right.process(originalRightIn);
454
455 const T filterAirLeftOut = _filterAirLeft.process(originalLeftIn);
456 const T filterAirRightOut = _filterAirRight.process(originalRightIn);
457
458
459 // Write to output, applying filter position and mix level
460 leftSamples[index] = (
461 originalLeftIn * (1 - _mix)
462 + filter1LeftOut * (1 - blendFilterPosition) * _mix
463 + filter2LeftOut * blendFilterPosition * _mix
464 + filterAirLeftOut * _airGain * _mix
465 )
466 * _outputGain;
467
468 rightSamples[index] = (
469 originalRightIn * (1 - _mix)
470 + filter1RightOut * (1 - blendFilterPosition) * _mix
471 + filter2RightOut * blendFilterPosition * _mix
472 + filterAirRightOut * _airGain * _mix
473 )
474 * _outputGain;
475 }
476 }

484 template <typename T>
485 Vowel SongbirdFilterModule<T>::_calcVowelForFreqMode(double modAmount) {
486 // get the first and second vowels
487 Vowel tempVowel1(getVowelDescription(getVowel1()));
488 Vowel tempVowel2(getVowelDescription(getVowel2()));
489
490 Vowel retVal(tempVowel1);
491
492 for (size_t iii {0}; iii < NUM_FORMANTS_PER_VOWEL; iii++) {
493 // Calculate frequency modulation
494 const double freqDelta {tempVowel2[iii].frequency - tempVowel1[iii].frequency};
495 retVal[iii].frequency = tempVowel1[iii].frequency + freqDelta * (_filterPosition + modAmount);
496
497 // Calculate gain modulation
498 const double gainDelta {tempVowel2[iii].gaindB - tempVowel1[iii].gaindB};
499 retVal[iii].gaindB = tempVowel1[iii].gaindB + gainDelta * (_filterPosition + modAmount);
500 }
501
502 return retVal;
503 }
